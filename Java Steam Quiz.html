        function loadQuestion() {

            const q = myQuestions[currentIdx];

            ui.scenario.textContent = q.scenario;

            ui.code.innerHTML = `${q.codeBefore}<span class="highlight-blank">____</span>${q.codeAfter}`;

        }


        function handleComplete(isCorrect) {

            ui.interaction.classList.add('hidden');

            ui.nextSection.classList.remove('hidden');

            }

        }


        function checkAnswer() {

            const userAns = ui.input.value.trim().toLowerCase();

            const correctAns = myQuestions[currentIdx].method.toLowerCase();


            if (userAns === correctAns) {

                score += 10;

                handleComplete(true);

            } else {

                ui.feedback.textContent = ``; // 힌트 제거됨

                ui.feedback.className = 'h-6';

                ui.input.classList.add('border-red-500');

                setTimeout(() => ui.input.classList.remove('border-red-500'), 500);

            }

        }


        function next() {

            currentIdx++;

            if (currentIdx < totalToPlay) loadQuestion();

            else finish();

        }


        function finish() {

            ui.quizCard.classList.add('hidden');

            ui.resultScreen.classList.remove('hidden');

            ui.finalScore.textContent = score;

        }


        ui.hint.onclick = () => ui.modal.classList.remove('hidden');

        ui.closeModal.onclick = () => ui.modal.classList.add('hidden');

        ui.revealAnswer.onclick = () => {

            ui.modal.classList.add('hidden');

            handleComplete(false);

        };


        ui.submit.onclick = checkAnswer;

        ui.nextBtn.onclick = next;

        ui.input.onkeypress = (e) => {

            if (e.key === 'Enter') checkAnswer();

        };


        window.onload = initQuiz;

    </script>
