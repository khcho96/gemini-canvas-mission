        function safeJsonParse(str) {

            try {

                return JSON.parse(str);

            } catch (e) {

                const jsonMatch = str.match(/\{[\s\S]*\}/);

                if (!jsonMatch) throw e;

                let sanitized = jsonMatch[0].replace(/\\(?![nrbtu"\\\/])/g, "\\\\");

                return JSON.parse(sanitized);

            }

        }

        async function callGemini(userQuery, systemPrompt = "", isJson = false, useSearch = false) {

            const endpoint = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            const payload = {

                contents: [{ parts: [{ text: userQuery }] }],

                systemInstruction: systemPrompt ? { parts: [{ text: systemPrompt }] } : undefined,

                generationConfig: isJson ? { responseMimeType: "application/json" } : {}

            };

            if (useSearch) payload.tools = [{ "google_search": {} }];


            const delays = [1000, 2000, 4000];

            for (let i = 0; i < 4; i++) {

                try {

                    const response = await fetch(endpoint, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    if (!response.ok) {

                        if (response.status === 401 && useSearch) return await callGemini(userQuery, systemPrompt, isJson, false);

                        throw new Error(`HTTP ${response.status}`);

                    }

                    const result = await response.json();

                    const text = result.candidates?.[0]?.content?.parts?.[0]?.text || "";

                    return text;

                } catch (error) {

                    if (i < 3) await new Promise(r => setTimeout(r, delays[i]));

                    else throw error;

                }

            }

        }

            JSON 응답 시 백슬래시는 반드시 이중 백슬래시(\\\\)로 처리하여 유효한 형식을 유지하십시오.`;


            // 프롬프트를 수정하여 설명, 입력, 출력을 별도의 JSON 필드로 요구합니다.

            const user = `실제로 존재하는 ${levelDesc} 문제를 찾아 다음 JSON으로만 응답하세요:

            {

                "platform": "사이트명",

                "origin_level": "실제 티어/레벨 (예: 실버3, Lv2)",

                "title": "실제 제목",

                "url": "정확한 URL",

                "description": "문제 설명 원문 100% 복사 (요약/설명 추가 절대 금지, 수식은 $, 줄바꿈은 \\n 사용)",

                "correct_answer": "6개 중 정답"

            }`;


            try {

                const response = await callGemini(user, sys, true, true);

                currentProblem = safeJsonParse(response);

            } catch (error) {

        function renderQuiz() {

            document.getElementById('problem-platform').innerText = currentProblem.platform;

            document.getElementById('problem-level-badge').innerText = currentProblem.origin_level;

            document.getElementById('problem-title').innerText = currentProblem.title;

            document.getElementById('problem-link').href = currentProblem.url;

            

            // 분리된 내용을 각각의 요소에 렌더링합니다.

            renderMarkdown('problem-description', currentProblem.description || "문제 설명이 제공되지 않았습니다.");

            renderMarkdown('problem-input', currentProblem.input || "입력 조건이 제공되지 않았습니다.");

            renderMarkdown('problem-output', currentProblem.output || "출력 조건이 제공되지 않았습니다.");


            const grid = document.getElementById('option-grid');

            grid.innerHTML = '';

            selectedOption = null;


            currentProblem.options.forEach(opt => {

                const btn = document.createElement('button');

                btn.className = "option-btn p-4 border-2 border-slate-100 rounded-2xl text-[14px] bg-white hover:border-blue-300 transition-all text-left font-bold shadow-sm active:bg-blue-50 text-slate-700";

                btn.innerText = opt;

                btn.onclick = () => {

                    document.querySelectorAll('.option-btn').forEach(b => {

                        b.classList.remove('bg-blue-600', 'text-white', 'border-blue-600');

                        b.classList.add('bg-white', 'text-slate-700', 'border-slate-100');

                    });

                    btn.classList.remove('bg-white', 'text-slate-700', 'border-slate-100');

                    btn.classList.add('bg-blue-600', 'text-white', 'border-blue-600');

                    selectedOption = opt;

                };

                grid.appendChild(btn);

            });


            showStep('step-quiz');

        }


        async function checkAnswer() {

            if (!selectedOption) { alert("전략을 선택하세요!"); return; }

            showStep('loading');

            document.getElementById('loading-text').innerText = "Gemini가 전략을 정밀 분석 중입니다...";


            const query = `사용자 전략: ${selectedOption}. 정답: ${currentProblem.correct_answer}. 문제: ${currentProblem.title}. 

            다음 JSON 형식으로 상세 리포트를 작성하세요 (수식은 $ 사용, 줄바꿈은 \\n 사용, 백슬래시 이중 처리):

            {

                "is_correct": boolean,

                "judgment": "마크다운 판단 근거 (N의 범위에 따른 시간복잡도 포함)",

                "strategy": "단계별 알고리즘 설계 프로세스",

                "concept": "핵심 알고리즘 개념 요약"

            }`;


            try {

                const response = await callGemini(query, "알고리즘 분석 전문가로서 답변하세요. JSON 규격을 엄격히 준수하세요.", true);

                const report = safeJsonParse(response);

                

                if (!solvedProblemTitles.includes(currentProblem.title)) {

                    solvedProblemTitles.push(currentProblem.title);

                }


                document.getElementById('result-problem-title').innerText = currentProblem.title;

                document.getElementById('result-problem-link').href = currentProblem.url;


                const statusDiv = document.getElementById('result-status');

                statusDiv.className = `text-center p-8 rounded-[2rem] border-2 ${report.is_correct ? 'bg-green-50 border-green-200 text-green-800' : 'bg-orange-50 border-orange-200 text-orange-800'}`;

                statusDiv.innerHTML = `<div class="text-5xl mb-4">${report.is_correct ? '🎯' : '💡'}</div><h3 class="text-2xl font-black">${report.is_correct ? '정확한 전략입니다!' : '전략 수정이 필요합니다.'}</h3><p class="text-sm mt-3 font-bold opacity-75">선택: ${selectedOption} | 최적: ${currentProblem.correct_answer}</p>`;


                renderMarkdown('exp-reason', report.judgment);

                renderMarkdown('exp-strategy', report.strategy);

                renderMarkdown('exp-concept', report.concept);


                showStep('step-result');

            } catch (error) {

                console.error(error);

                alert("리포트 생성 오류");

                showStep('step-quiz');

            }

        }

            currentProblem = null;

            showStep('step-level');

        }
