    <script>

        const apiKey = ""; // API Key is provided by the environment

        let myIngredients = [];

        let generatedRecipes = [];


        function addIngredient() {

            const input = document.getElementById('ingredientInput');

            const value = input.value.trim();

            if (!value) return;


            const items = value.split(',').map(i => i.trim()).filter(i => i !== "");

            items.forEach(item => {

                if (!myIngredients.includes(item)) myIngredients.push(item);

            });


            input.value = "";

            renderIngredients();

        }


        function removeIngredient(name) {

            myIngredients = myIngredients.filter(i => i !== name);

            renderIngredients();

        }


        function renderIngredients() {

            const list = document.getElementById('ingredientList');

            if (myIngredients.length === 0) {

                list.innerHTML = `<p class="text-slate-300 text-sm italic">재료를 한 개 이상 입력해주세요.</p>`;

                return;

            }

            list.innerHTML = myIngredients.map(item => `

                <div class="bg-orange-50 text-orange-600 px-4 py-2 rounded-xl flex items-center gap-2 border border-orange-100 shadow-sm animate-in fade-in slide-in-from-bottom-1">

                    <span class="text-sm font-bold">${item}</span>

                    <button onclick="removeIngredient('${item}')" class="hover:text-red-500 font-bold text-xl leading-none">&times;</button>

                </div>

            `).join('');

        }


        async function fetchWithRetry(url, options, retries = 5, backoff = 1000) {

            try {

                const response = await fetch(url, options);

                if (!response.ok) throw new Error('API request failed');

                return await response.json();

            } catch (err) {

                if (retries > 0) {

                    await new Promise(resolve => setTimeout(resolve, backoff));

                    return fetchWithRetry(url, options, retries - 1, backoff * 2);

                }

                throw err;

            }

        }


        async function generateAiRecipes() {

            if (myIngredients.length === 0) {

                alert("냉장고에 있는 재료를 먼저 입력해주세요!");

                return;

            }


            const loadingView = document.getElementById('loadingView');

            const resultSection = document.getElementById('resultSection');

            const generateBtn = document.getElementById('generateBtn');


            loadingView.classList.remove('hidden');

            resultSection.classList.add('hidden');

            generateBtn.disabled = true;

            generateBtn.classList.add('opacity-50');


            const systemPrompt = `You are a professional chef AI agent. Based on the user's available ingredients, suggest exactly 8 creative and delicious recipes across various cuisines (Korean, Chinese, Japanese, Western, Southeast Asian, etc.).

            The response must be in JSON format with the following structure:

            {

                "recipes": [

                    {

                        "id": number,

                        "name": "Recipe Name",

                        "category": "Cuisine Category",

                        "difficulty": "Easy/Medium/Hard",

                        "time": number, // Output time as a number only, without "분"

                        "availableIngredientsUsed": ["list", "of", "input", "ingredients"],

                        "additionalIngredientsNeeded": ["minimal", "extra", "items"],

                        "fullIngredientsList": ["detailed", "list", "with", "amounts"],

                        "steps": ["step1", "step2 with **bold** highlights", "..."],

                        "description": "Brief delicious description"

                    }

                ]

            }

            Use Korean for all text content except keys. For the steps, use markdown **bold** to highlight important cooking techniques or heat levels (e.g., **강불**, **약불**, **10분간**).`;


            const userQuery = `냉장고 재료: ${myIngredients.join(', ')}. 기본 조미료(물, 소금, 설탕, 간장, 식용유 등)는 이미 구비되어 있다고 가정하고 총 8개의 레시피를 제안해줘.`;


            try {

                const result = await fetchWithRetry(

                    `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`,

                    {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify({

                            contents: [{ parts: [{ text: userQuery }] }],

                            systemInstruction: { parts: [{ text: systemPrompt }] },

                            generationConfig: { responseMimeType: "application/json" }

                        })

                    }

                );


                const data = JSON.parse(result.candidates[0].content.parts[0].text);

                generatedRecipes = data.recipes;

                renderRecipeCards(generatedRecipes);

                resultSection.classList.remove('hidden');

            } catch (error) {

                console.error(error);

                alert("레시피 생성 중 오류가 발생했습니다. 잠시 후 다시 시도해주세요.");

            } finally {

                loadingView.classList.add('hidden');

                generateBtn.disabled = false;

                generateBtn.classList.remove('opacity-50');

                resultSection.scrollIntoView({ behavior: 'smooth' });

            }

        }


        function renderRecipeCards(recipes) {

            const grid = document.getElementById('recipeGrid');

            grid.innerHTML = recipes.map(recipe => `

                <div class="recipe-card bg-white p-6 rounded-3xl border border-slate-100 shadow-sm flex flex-col h-full" onclick="showDetail(${recipe.id})">

                    <div class="flex justify-between items-start mb-4">

                        <div>

                            <span class="text-[10px] font-bold text-orange-500 uppercase tracking-wider mb-1 block">${recipe.category}</span>

                            <h3 class="text-xl font-bold text-slate-800 leading-tight">${recipe.name}</h3>

                        </div>

                        <span class="px-2 py-1 rounded-lg text-[10px] font-bold ${recipe.difficulty === '쉬움' ? 'bg-green-50 text-green-600' : 'bg-amber-50 text-amber-600'}">

                            ${recipe.difficulty === 'Easy' ? '쉬움' : (recipe.difficulty === 'Medium' ? '보통' : (recipe.difficulty === 'Hard' ? '어려움' : recipe.difficulty))}

                        </span>

                    </div>

                    <p class="text-slate-500 text-xs mb-6 flex-1">${recipe.description}</p>

                    <div class="flex items-center justify-between mt-auto pt-4 border-t border-slate-50">

                        <span class="text-slate-400 text-xs font-medium">⏱️ ${recipe.time}분</span>

                        <div class="flex -space-x-1">

                            ${recipe.availableIngredientsUsed.slice(0, 3).map(() => `<div class="w-5 h-5 rounded-full bg-orange-100 border-2 border-white"></div>`).join('')}

                        </div>

                    </div>

                </div>

            `).join('');

        }


        function formatMarkdownBold(text) {

            return text.replace(/\*\*(.*?)\*\*/g, '<strong class="markdown-bold">$1</strong>');

        }


        function showDetail(id) {

            const recipe = generatedRecipes.find(r => r.id === id);

            if (!recipe) return;


            document.getElementById('modalTitle').innerText = recipe.name;

            document.getElementById('modalCategory').innerText = recipe.category;

            document.getElementById('modalDiff').innerText = recipe.difficulty === 'Easy' ? '쉬움' : (recipe.difficulty === 'Medium' ? '보통' : (recipe.difficulty === 'Hard' ? '어려움' : recipe.difficulty));

            document.getElementById('modalTime').innerText = `${recipe.time}분`;

            

            const colors = ['#ffedd5', '#fef3c7', '#ecfccb', '#dcfce7', '#f1f5f9', '#e0e7ff', '#fae8ff', '#f1f2f6'];

            document.getElementById('modalHeaderImg').style.backgroundColor = colors[id % colors.length];


            const ingList = document.getElementById('modalIngredients');

            ingList.innerHTML = recipe.fullIngredientsList.map(ing => `

                <li class="flex items-center gap-2">

                    <span class="w-1 h-1 bg-slate-300 rounded-full"></span>

                    ${ing}

                </li>

            `).join('');


            const stepsContainer = document.getElementById('modalSteps');

            stepsContainer.innerHTML = recipe.steps.map((step, idx) => `

                <div class="flex gap-4 group">

                    <span class="shrink-0 w-6 h-6 bg-slate-100 text-slate-500 rounded-lg flex items-center justify-center font-bold text-[10px] group-hover:bg-orange-500 group-hover:text-white transition-colors">${idx + 1}</span>

                    <p class="text-slate-700 leading-relaxed">${formatMarkdownBold(step)}</p>

                </div>

            `).join('');


            document.getElementById('recipeModal').classList.remove('hidden');

            document.body.style.overflow = 'hidden';

        }


        function closeModal() {

            document.getElementById('recipeModal').classList.add('hidden');

            document.body.style.overflow = 'auto';

        }


        document.getElementById('ingredientInput').addEventListener('keypress', (e) => {

            if (e.key === 'Enter') addIngredient();

        });


        window.onclick = function(event) {

            if (event.target == document.getElementById('recipeModal')) closeModal();

        }


        renderIngredients();

    </script>
