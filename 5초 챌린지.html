        <div id="finalTime" style="font-size: 1.1rem;">기록: 0.00초</div>

        <hr style="border: 0; border-top: 1px solid #ccc; margin: 10px 0;">

<script>

    let startTime;

    const TARGET_TIME = 5000; // 5초 (ms)


    const startBtn = document.getElementById('startBtn');

    const stopBtn = document.getElementById('stopBtn');

    const resetBtn = document.getElementById('resetBtn');

    const timerDisplay = document.getElementById('timerDisplay');

    const statusText = document.getElementById('status');

    const resultPanel = document.getElementById('resultPanel');

    const finalTimeText = document.getElementById('finalTime');

    const errorDisplayText = document.getElementById('errorDisplay');

    const totalScoreText = document.getElementById('totalScore');

    const scoreLabel = document.getElementById('scoreLabel');


    // 시작 버튼 클릭

    startBtn.addEventListener('click', () => {

        startTime = Date.now();

        startBtn.style.display = 'none';

        stopBtn.style.display = 'block';

        statusText.innerText = "시간이 흐르고 있습니다...";

        statusText.style.color = "#e67e22";

        timerDisplay.innerText = "???"; 

        

        resultPanel.style.display = 'none';

        totalScoreText.classList.remove('perfect-score');

    });


    // 정지 버튼 클릭

    stopBtn.addEventListener('click', () => {

        const endTime = Date.now();

        const elapsed = endTime - startTime;

        

        stopBtn.style.display = 'none';

        resetBtn.style.display = 'block';

        

        calculateResult(elapsed);

    });


    // 다시 시작

    resetBtn.addEventListener('click', () => {

        resetBtn.style.display = 'none';

        startBtn.style.display = 'block';

        statusText.innerText = "준비되셨나요?";

        statusText.style.color = "#2c3e50";

        timerDisplay.innerText = "0.00";

        resultPanel.style.display = 'none';

        totalScoreText.classList.remove('perfect-score');

    });


    function calculateResult(elapsed) {

        // 소수점 둘째 자리까지만 계산에 반영하기 위해 반올림 처리

        const processedElapsed = Math.round(elapsed / 10) * 10;

        const diff = Math.abs(TARGET_TIME - processedElapsed);

        

        const secondsDisplay = (processedElapsed / 1000).toFixed(2);

        const diffDisplay = (diff / 1000).toFixed(2);

        

        // 결과 표시 기본 설정

        statusText.innerText = "결과 발표!";

        statusText.style.color = "#2ecc71";

        timerDisplay.innerText = secondsDisplay;

        

        finalTimeText.innerText = `기록: ${secondsDisplay}초`;

        errorDisplayText.innerText = `오차: ${diffDisplay}s`;

        

        // 점수 계산 (최대 100점, 오차 10ms당 1점씩 감점)

        let score = Math.max(0, 100 - (diff / 10));

        if (diff > 2000) score = 0; 

        

        const roundedScore = Math.round(score);

        totalScoreText.innerText = `${roundedScore}점`;

        

        // 코멘트 및 특별 연출

        let comment = "";

        if (diff === 0) {

            comment = "🎊 신의 경지! 완벽하게 5초입니다! 🎊";

            statusText.innerText = "PERFECT!!!";

            statusText.style.color = "var(--perfect-color)";

            totalScoreText.classList.add('perfect-score');

        } else if (diff <= 50) {

            comment = "와우! 인간 메트로놈이신가요? 🏆";

        } else if (diff <= 200) {

            comment = "거의 정확해요! 대단합니다! 👍";

        } else if (diff <= 500) {

            comment = "나쁘지 않은 감각이에요! 😊";

        } else if (diff <= 1000) {

            comment = "조금 더 집중해볼까요? 🧐";

        } else {

            comment = "시간이 너무 빠르거나 느리게 가네요! 😂";

        }

        

        scoreLabel.innerText = comment;

        resultPanel.style.display = 'block';

    }
